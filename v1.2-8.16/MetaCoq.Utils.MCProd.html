Lemma on_snd_on_snd {A B C D} (f : C → D) (g : B → C) (d : A × B) :

Lemma on_snd_eq_spec {A B C} (f g : B → C) (x : A × B) :
  f (snd x) = g (snd x) ↔

Proof.
  destruct x; unfold on_snd; cbn. split; congruence.
Qed.
